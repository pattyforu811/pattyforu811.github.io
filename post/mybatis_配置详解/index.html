<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE configuration
        PUBLIC "-//mybatis.org//DTD Config 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-config.dtd">
<configuration>
    <!-- 属性标签，引入外部的 properties配置文件的内容
        resource：引入类路径下的资源
        url：引入网络路径或者磁盘路径下的资源
    -->
    <properties resource="jdbc.properties"></properties>

    <!-- settings标签：用来设置每一个设置项  name:设置项的名字 value：值 -->
    <settings>
        <!-- 开启驼峰命名法规则，例如：数据库字段为emp_name 实体类属性为empName可以自动进行匹配 -->
        <setting name="mapUnderscoreToCamelCase" value="true"/>
		<!--开启懒加载-->
        <setting name="lazyLoadingEnabled" value="true"/>
        <!--false 按需加载属性-->
        <setting name="aggressiveLazyLoading" value="false"/>
        <!--开启已经配置的缓存(二级缓存) 默认true-->
        <setting name="cacheEnabled" value="true"/>
    </settings>

    <!-- 给java类型别名：type:指定要起别名的类型全类名
    默认别名：类名首字母小写
    指定别名：alias
    用途: 在mapper文件中返回值resultType可以使用别名-->
    <typeAliases>
        <!-- <typeAlias type="com.pattyforu.domain.Employee" alias="employee"/>-->

        <!-- package:为某个包下的所有类批量起别名 name:指定包名-->
        <!-- 批量起别名的情况下，
            如果出现默认的别名冲突的时候,
            可以通过使用 @Alias 注解配置为某个类型指定新的别名 ,别名不区分大小写-->
        <package name="com.pattyforu.domain"/>
    </typeAliases>

    <!-- 环境，mybatis可以配置多种环境 default：指定某种环境，可以达到快速切换环境的效果-->
    <environments default="mysql">
        <!-- mysql开发环境 -->
        <environment id="mysql">
            <!-- 事务管理器 type：事务管理器的类型-->
            <transactionManager type="JDBC"/>
            <!-- 数据源：type:数据源类型 -->
            <dataSource type="POOLED">
<!--            <property name="driver" value="com.mysql.jdbc.Driver"/>
                <property name="url" value="jdbc:mysql:///mybatis"/>
                <property name="username" value="root"/>
                <property name="password" value="root"/>-->
                <property name="driver" value="${jdbc.driverClass}"/>
                <property name="url" value="${jdbc.jdbcUrl}"/>
                <property name="username" value="${jdbc.user}"/>
                <property name="password" value="${jdbc.password}"/>
            </dataSource>
        </environment>

        <!-- oracle开发环境 -->
        <environment id="oracle"><!-- id="oracle"代表当前环境的唯一标识 -->
            <transactionManager type="JDBC"/>
            <dataSource type="POOLED">
                <property name="driver" value="${jdbc.orcl.driver}"/>
                <property name="url" value="${jdbc.orcl.url}"/>
                <property name="username" value="${jdbc.orcl.user}"/>
                <property name="password" value="${jdbc.orcl.password}"/>
            </dataSource>
        </environment>
    </environments>

    <!-- 数据库厂商标识 type用于指定数据库标识 type="DB_VENDOR":固定写法 -->
    <databaseIdProvider type="DB_VENDOR">
        <!-- 为不同的数据库厂商标识别名 ,
        sql配置文件中(Mapper文件中)在标签上指定databaseId="标识别名"配合使用,
        例如databaseId="mysql"则指定这条sql语句只有在mysql环境下才能被执行-->
        <property name="MySQL" value="mysql"/>
        <property name="Oracle" value="oracle"/>
    </databaseIdProvider>


    <!--将sql映射文件配置到全局配置文件中-->
    <mappers>
        <!-- mapper:指定一个sql映射文件
                    resource：引用类路径下的sql文件
                    url：引入网络路径或者磁盘路径下的资源

                    class:引用接口全类名(类的用 . 分隔)
                            1.有sql映射文件，文件名必须要和接口名一致，并且还需要与接口在同一目录下
                                解决办法:在resources下创建和接口相同的目录
                                         编译后就可以在target里看到接口和映射文件在相同目录下
                            2.没有映射文件，所有的sql语句可以利用注解写在接口方法上(不推荐使用)
                                @Select("select * from employee where id = #{id}")
         -->
        <!--<mapper resource="com.pattyforu.dao/EmployeeMapper.xml"/>-->
        <!--<mapper class="com.pattyforu.dao.EmployeeMapper"/>-->
        <!-- 批量配置 -->
        <package name="com.pattyforu.dao"/>
    </mappers>
</configuration>
