前置通知：@Before
后置通知：@AfterReturning
异常通知：@AfterThorwing
最终通知：@After
环绕通知：@Around

@Component("logUtils")
@Aspect
public class LogUtils {

    @Pointcut("execution(* com.pattyforu.service.impl.*.*(..))")
    private void pt(){}
    
    /*
     用于输出日志记录，想要在切入点（业务层方法）方法执行之前执行
     */
    @Before("pt()")
    public void logStart(){
        System.out.println("LogUtils的logStart方法开始记录日志了");
    }
    /**
     * 后置,想要在切入点（业务层方法）方法执行之后执行
     */
    @AfterReturning("pt()")
    public void logEnd(){
        System.out.println("LogUtils的logEnd方法开始记录日志了");
    }
    
    /**
     * 异常，想要在切入点（业务层方法）方法发生异常之后执行
     */
    @AfterThrowing("pt()")
    public void logException(){
        System.out.println("LogUtils的logException方法开始记录日志了");
    }
    
    /**
     * 最终，想要在切入点（业务层方法）是否正常执行都会在最后执行
     */
    @After("pt()")
    public void logFianl(){
        System.out.println("LogUtils的logFianl方法开始记录日志了");
    }
    /**
     * 环绕通知
     */
//    @Around("pt()")
    public Object logAround(ProceedingJoinPoint pjp){
        Object obj = null;
        try {
            Object[] objs = pjp.getArgs();//得到方法执行的所需参数
             System.out.println("前置");
             obj = pjp.proceed(objs);//明确调用切入点（业务层）方法
             System.out.println("后置");
            return obj;
        } catch (Throwable e) {
            System.out.println("异常");
            throw new RuntimeException();
        }finally {
            System.out.println("最终");
        }
        
    }
